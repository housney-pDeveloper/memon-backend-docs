LoginController API 문서

소스 경로: biz/login/controller/LoginController.java

참고: 모든 API는 Gateway를 통해 접근합니다. {clientType}은 클라이언트 유형에 따라 app 또는 web으로 대체합니다.

보안 요구사항

  • 비밀번호는 반드시 RSA 공개키로 암호화하여 전송해야 합니다
  • 평문 비밀번호 전송 시 복호화 오류가 발생합니다
  • RSA 공개키는 AuthController의 getPublicKey API를 통해 조회합니다

2단계 로그인 프로세스 가이드

  1. POST /{clientType}/{version}/auth/getPublicKey API를 호출하여 RSA 공개키를 조회합니다
  2. 조회된 공개키로 비밀번호를 RSA 암호화합니다 (JSEncrypt 또는 WebCrypto API 사용)
  3. POST /{clientType}/{version}/login API를 호출합니다
    • 단일 고객사: type=LOGIN_COMPLETE, JWT 토큰 직접 발급
    • 복수 고객사: type=CLIENT_SELECT, tempToken + clientList 반환
  4. 복수 고객사인 경우, POST /{clientType}/{version}/login/selectClient API로 고객사 선택 후 JWT 토큰 발급
  5. 약관 재동의 체크: 응답의 termsReagreeRequired가 true인 경우, reagreeTermsList에 있는 약관에 대해 재동의 필요
    • 약관 동의 API: POST /{clientType}/{version}/terms/agreeTerms
    • 약관 상세 조회: TermsController 참조

/** 로그인 요청 */
interface LoginRequest {
  /** 이메일 */
  email: string;
  /** 암호화된 비밀번호 (RSA 공개키로 암호화, Base64 인코딩) */
  encryptedPassword: string;
}

/** 로그인 결과 타입 */
type LoginResultType = 'LOGIN_COMPLETE' | 'CLIENT_SELECT';

/** 고객사 목록 아이템 */
interface LoginClientItem {
  /** 사용자-고객사 관계번호 */
  userClientNo: number;
  /** 사용자 번호 */
  userNo: number;
  /** 사용자 코드 */
  userCode: string;
  /** 고객사 번호 */
  clientNo: number;
  /** 고객사 코드 */
  clientCode: string;
  /** 고객사명 */
  clientName: string;
  /** 사용자 역할 */
  userRole: string;
}

/** 재동의 필요 약관 항목 */
interface TermsReagreeItem {
  /** 약관 번호 */
  termsNo: number;
  /** 약관 코드 */
  termsCode: string;
  /** 약관 명 */
  termsName: string;
  /** 약관 버전 */
  version: string;
  /** 필수 여부 */
  requiredYn: string;
  /** 약관 전문 여부 (Y: 전문 있음, N: 제목만) */
  contentYn: string;
}

/** 로그인 응답 */
interface LoginResponse {
  /** 로그인 결과 타입 */
  type: LoginResultType;
  /** 사용자 번호 */
  userNo?: number;
  /** 사용자 코드 */
  userCode?: string;
  /** 이메일 */
  email?: string;
  /** 사용자명 */
  userName?: string;
  /** 임시 토큰 (CLIENT_SELECT 시) */
  tempToken?: string;
  /** 임시 토큰 만료 시간(초) */
  tempTokenExpiresInSeconds?: number;
  /** 고객사 목록 (CLIENT_SELECT 시) */
  clientList?: LoginClientItem[];
  /** 사용자-고객사 관계번호 (LOGIN_COMPLETE 시) */
  userClientNo?: number;
  /** 고객사 번호 (LOGIN_COMPLETE 시) */
  clientNo?: number;
  /** 고객사 코드 (LOGIN_COMPLETE 시) */
  clientCode?: string;
  /** 고객사명 (LOGIN_COMPLETE 시) */
  clientName?: string;
  /** 권한 (LOGIN_COMPLETE 시) */
  userRole?: string;
  /** 액세스 토큰 (LOGIN_COMPLETE 시) */
  accessToken?: string;
  /** 리프레시 토큰 (LOGIN_COMPLETE 시) */
  refreshToken?: string;
  /** 만료 시간(초) (LOGIN_COMPLETE 시) */
  expiresInSeconds?: number;
  /** 발급 일시 */
  issuedAt?: string;
  /** 약관 재동의 필요 여부 (LOGIN_COMPLETE 시) */
  termsReagreeRequired?: boolean;
  /** 재동의 필요 약관 목록 (있는 경우만) */
  reagreeTermsList?: TermsReagreeItem[];
}

목적

이미 로그인된 사용자가 다른 고객사로 전환합니다.

토큰 처리

  • 기존 ACCESS 토큰은 블랙리스트에 등록됩니다
  • 기존 REFRESH 토큰은 Redis에서 삭제됩니다
  • 새로운 ACCESS/REFRESH 토큰이 발급됩니다

주의사항

  • TEMP 토큰으로는 호출 불가 (Gateway에서 거부)
  • 프론트엔드에서 토큰 교체 필수

// ---------- Response Interfaces ----------

/** 로그인 결과 타입 */
type LoginResultType = 'LOGIN_COMPLETE' | 'CLIENT_SELECT';

/** 고객사 목록 아이템 (로그인 응답용) */
interface LoginClientItem {
  /** 사용자-고객사 관계번호 */
  userClientNo: number;
  /** 사용자 번호 */
  userNo: number;
  /** 사용자 코드 */
  userCode: string;
  /** 고객사 번호 */
  clientNo: number;
  /** 고객사 코드 */
  clientCode: string;
  /** 고객사명 */
  clientName: string;
  /** 사용자 역할 */
  userRole: string;
}

/** 재동의 필요 약관 항목 */
interface TermsReagreeItem {
  /** 약관 번호 */
  termsNo: number;
  /** 약관 코드 */
  termsCode: string;
  /** 약관 명 */
  termsName: string;
  /** 약관 버전 */
  version: string;
  /** 필수 여부 */
  requiredYn: string;
  /** 약관 전문 여부 (Y: 전문 있음, N: 제목만) */
  contentYn: string;
}

/** 로그인 응답 */
interface LoginResponse {
  /** 로그인 결과 타입 */
  type: LoginResultType;
  /** 사용자 번호 */
  userNo?: number;
  /** 사용자 코드 */
  userCode?: string;
  /** 이메일 */
  email?: string;
  /** 사용자명 */
  userName?: string;
  /** 임시 토큰 (CLIENT_SELECT 시) */
  tempToken?: string;
  /** 임시 토큰 만료 시간(초) */
  tempTokenExpiresInSeconds?: number;
  /** 고객사 목록 (CLIENT_SELECT 시) */
  clientList?: LoginClientItem[];
  /** 사용자-고객사 관계번호 (LOGIN_COMPLETE 시) */
  userClientNo?: number;
  /** 고객사 번호 (LOGIN_COMPLETE 시) */
  clientNo?: number;
  /** 고객사 코드 (LOGIN_COMPLETE 시) */
  clientCode?: string;
  /** 고객사명 (LOGIN_COMPLETE 시) */
  clientName?: string;
  /** 권한 (LOGIN_COMPLETE 시) */
  userRole?: string;
  /** 액세스 토큰 (LOGIN_COMPLETE 시) */
  accessToken?: string;
  /** 리프레시 토큰 (LOGIN_COMPLETE 시) */
  refreshToken?: string;
  /** 만료 시간(초) (LOGIN_COMPLETE 시) */
  expiresInSeconds?: number;
  /** 발급 일시 */
  issuedAt?: string;
  /** 약관 재동의 필요 여부 (LOGIN_COMPLETE 시) */
  termsReagreeRequired?: boolean;
  /** 재동의 필요 약관 목록 (있는 경우만) */
  reagreeTermsList?: TermsReagreeItem[];
}

/** 고객사 선택 응답 */
type ClientSelectResponse = LoginResponse;

/** 고객사 전환 응답 */
type ClientSwitchResponse = LoginResponse;

/** 현재 로그인 사용자 정보 */
interface CurrentUser {
  /** 사용자 번호 */
  userNo: number;
  /** 사용자-고객사 관계번호 */
  userClientNo: number;
  /** 사용자 코드 */
  userCode: string;
  /** 이메일 */
  email: string;
  /** 사용자명 */
  userName: string;
  /** 성별 */
  gender?: string;
  /** 권한 */
  userRole: string;
}

/** 현재 선택된 고객사 정보 */
interface CurrentClient {
  /** 고객사 번호 */
  clientNo: number;
  /** 고객사 코드 */
  clientCode: string;
  /** 고객사명 */
  clientName: string;
}

/** 고객사 목록 아이템 (컨텍스트용) */
interface ClientItem {
  /** 고객사 번호 */
  clientNo: number;
  /** 고객사 코드 */
  clientCode: string;
  /** 고객사명 */
  clientName: string;
  /** 현재 선택된 고객사 여부 */
  current: boolean;
}

/** 사용자 컨텍스트 응답 */
interface UserContextResponse {
  /** 현재 로그인 사용자 정보 */
  currentUser: CurrentUser;
  /** 현재 선택된 고객사 정보 */
  currentClient: CurrentClient;
  /** 사용자가 속한 고객사 목록 */
  clientList: ClientItem[];
}
