🔐 프론트엔드 로그인 암호화 보안 가이드

1. 개요

이 문서는 React 기반 프론트엔드에서 로그인 시 비밀번호를 RSA 공개키로 암호화하여 전송하는 방법을 설명합니다.

⚠️ 중요: 이 암호화는 "보안 강화"가 아닌 "전송 중 노출 최소화" 목적입니다.
HTTPS(TLS)가 전제되어야 하며, 암호화만으로 보안이 완성되지 않습니다.

// src/api/http.ts
import axios, { AxiosInstance, AxiosRequestConfig } from 'axios';
import JSEncrypt from 'jsencrypt';

// API 기본 URL 설정
const BASE_URL = process.env.REACT_APP_API_URL || 'https://api.example.com';

// Axios 인스턴스 생성
const axiosInstance: AxiosInstance = axios.create({
  baseURL: BASE_URL,
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json',
  },
});

// 응답 인터셉터: 공통 에러 처리
axiosInstance.interceptors.response.use(
  (response) => response,
  (error) => {
    // 에러 처리 로직
    console.error('API Error:', error.response?.data || error.message);
    return Promise.reject(error);
  }
);

export default axiosInstance;

4-2. RSA 암호화 유틸리티

// src/utils/crypto.ts
import JSEncrypt from 'jsencrypt';

/**
 * RSA 공개키 저장소 (메모리에만 저장)
 * ⚠️ LocalStorage에 저장하지 마세요!
 */
let cachedPublicKey: string | null = null;

/**
 * RSA 공개키 설정
 * @param publicKey Base64 인코딩된 RSA 공개키
 */
export const setPublicKey = (publicKey: string): void => {
  cachedPublicKey = publicKey;
};

/**
 * RSA 공개키 초기화 (로그아웃 시 호출)
 */
export const clearPublicKey = (): void => {
  cachedPublicKey = null;
};

/**
 * 저장된 공개키 존재 여부 확인
 */
export const hasPublicKey = (): boolean => {
  return cachedPublicKey !== null;
};

/**
 * RSA 공개키로 데이터 암호화
 * @param plainText 암호화할 평문
 * @returns Base64 인코딩된 암호문
 * @throws Error 공개키가 설정되지 않았거나 암호화 실패 시
 */
export const encryptWithPublicKey = (plainText: string): string => {
  if (!cachedPublicKey) {
    throw new Error('공개키가 설정되지 않았습니다. 먼저 공개키를 조회하세요.');
  }

  const encrypt = new JSEncrypt();

  // PEM 형식으로 변환
  const pemKey = `-----BEGIN PUBLIC KEY-----\n${cachedPublicKey}\n-----END PUBLIC KEY-----`;
  encrypt.setPublicKey(pemKey);

  const encrypted = encrypt.encrypt(plainText);

  if (!encrypted) {
    throw new Error('암호화에 실패했습니다.');
  }

  return encrypted;
};

4-3. http.postEnc 함수 구현

// src/api/http.ts (계속)
import { encryptWithPublicKey, hasPublicKey, setPublicKey } from '../utils/crypto';

/**
 * 공개키 조회 API 호출
 * @returns 공개키 정보
 */
export const fetchPublicKey = async (): Promise<void> => {
  try {
    const response = await axiosInstance.post('/web/v1/auth/getPublicKey');
    const { publicKey } = response.data.data;

    if (!publicKey) {
      throw new Error('공개키 응답이 유효하지 않습니다.');
    }

    setPublicKey(publicKey);
    console.debug('공개키 조회 완료');
  } catch (error) {
    console.error('공개키 조회 실패:', error);
    throw error;
  }
};

/**
 * 암호화된 비밀번호가 필요한 API 요청
 *
 * 사용법:
 * - 일반 POST 요청: http.post('/web/api/data', { field: 'value' })
 * - 암호화 POST 요청: http.postEnc('/web/api/login', { email: 'user@example.com' }, 'myPassword123')
 *
 * @param url API 엔드포인트
 * @param data 요청 데이터 (비밀번호 제외)
 * @param password 암호화할 비밀번호
 * @param config Axios 설정 (선택)
 */
export const postEnc = async <T>(
  url: string,
  data: object,
  password: string,
  config?: AxiosRequestConfig
): Promise<T> => {
  // 공개키가 없으면 먼저 조회
  if (!hasPublicKey()) {
    await fetchPublicKey();
  }

  // 비밀번호 RSA 암호화
  const encryptedPassword = encryptWithPublicKey(password);

  // 암호화된 비밀번호를 요청 데이터에 추가
  const requestData = {
    ...data,
    encryptedPassword,
  };

  const response = await axiosInstance.post<T>(url, requestData, config);
  return response.data;
};

// HTTP 메서드 객체 export
export const http = {
  get: axiosInstance.get,
  post: axiosInstance.post,
  put: axiosInstance.put,
  delete: axiosInstance.delete,
  postEnc,  // 암호화가 필요한 POST 요청
};

5. 로그인 구현 예시

5-1. 로그인 서비스

// src/services/authService.ts
import { http, fetchPublicKey } from '../api/http';
import { clearPublicKey } from '../utils/crypto';

interface LoginResponse {
  code: number;
  message: string;
  data: {
    email: string;
    accessToken: string;
    issuedAt: string;
  };
}

/**
 * 로그인 처리
 *
 * @param email 사용자 이메일
 * @param password 비밀번호 (평문 - 내부에서 암호화됨)
 * @returns 로그인 결과 (JWT 토큰 포함)
 */
export const login = async (
  email: string,
  password: string
): Promise<LoginResponse> => {
  try {
    // http.postEnc가 자동으로:
    // 1. 공개키 조회 (없으면)
    // 2. 비밀번호 RSA 암호화
    // 3. API 요청
    const response = await http.postEnc<LoginResponse>(
      '/web/v1/login',
      { email },
      password
    );

    // 토큰 저장 (httpOnly 쿠키 권장, 아래는 예시)
    if (response.data.accessToken) {
      sessionStorage.setItem('accessToken', response.data.accessToken);
    }

    return response;
  } catch (error) {
    console.error('로그인 실패:', error);
    throw error;
  }
};

/**
 * 로그아웃 처리
 */
export const logout = (): void => {
  // 토큰 삭제
  sessionStorage.removeItem('accessToken');

  // 캐시된 공개키 초기화
  clearPublicKey();
};

5-2. React 로그인 컴포넌트

// src/components/LoginForm.tsx
import React, { useState } from 'react';
import { login } from '../services/authService';

const LoginForm: React.FC = () => {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setLoading(true);
    setError(null);

    try {
      const result = await login(email, password);

      if (result.code === 1) {
        // 로그인 성공 - 메인 페이지로 이동
        window.location.href = '/dashboard';
      } else {
        setError(result.message);
      }
    } catch (err: any) {
      setError(err.response?.data?.message || '로그인에 실패했습니다.');
    } finally {
      setLoading(false);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label htmlFor="email">이메일</label>
        <input
          id="email"
          type="email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          required
          disabled={loading}
        />
      </div>

      <div>
        <label htmlFor="password">비밀번호</label>
        <input
          id="password"
          type="password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
          required
          disabled={loading}
          autoComplete="current-password"
        />
      </div>

      {error && <div className="error">{error}</div>}

      <button type="submit" disabled={loading}>
        {loading ? '로그인 중...' : '로그인'}
      </button>
    </form>
  );
};

export default LoginForm;

6. 보안 주의사항

🚫 절대 금지 사항

  • LocalStorage에 공개키 저장 금지: 공개키는 메모리에만 유지하세요.
  • AES 고정 키 하드코딩 금지: 프론트엔드에 대칭키를 저장하면 안 됩니다.
  • 비밀번호 평문 로깅 금지: console.log로 비밀번호를 출력하지 마세요.
  • Private Key 클라이언트 배포 금지: 개인키는 서버에만 존재해야 합니다.

⚠️ 주의 사항

  • 공개키는 세션당 또는 로그인 시도당 새로 조회하세요.
  • HTTPS(TLS) 연결이 전제되어야 합니다.
  • XSS 공격 방어가 선행되어야 합니다 (React의 기본 이스케이프 활용).
  • 개발 환경에서도 실제 비밀번호를 로그에 출력하지 마세요.

✅ 권장 사항

  • JWT 토큰은 httpOnly 쿠키에 저장하는 것을 권장합니다.
  • 로그인 실패 시 구체적인 오류 메시지 대신 일반 메시지를 표시하세요.
  • 비밀번호 입력 필드에 autoComplete="current-password" 속성을 추가하세요.
  • 공개키 조회 실패 시 적절한 재시도 로직을 구현하세요.

8. WebCrypto API 대안

JSEncrypt 대신 브라우저 내장 WebCrypto API를 사용할 수도 있습니다. WebCrypto는 더 나은 성능과 보안을 제공하지만, 구현이 더 복잡합니다.

// src/utils/webCrypto.ts
/**
 * WebCrypto API를 사용한 RSA 암호화
 *
 * @param publicKeyBase64 Base64 인코딩된 공개키
 * @param plainText 암호화할 평문
 * @returns Base64 인코딩된 암호문
 */
export const encryptWithWebCrypto = async (
  publicKeyBase64: string,
  plainText: string
): Promise<string> => {
  // Base64 디코딩
  const binaryDer = Uint8Array.from(
    atob(publicKeyBase64),
    (c) => c.charCodeAt(0)
  );

  // 공개키 임포트
  const publicKey = await crypto.subtle.importKey(
    'spki',
    binaryDer,
    {
      name: 'RSA-OAEP',
      hash: 'SHA-256',
    },
    false,
    ['encrypt']
  );

  // 평문을 바이트 배열로 변환
  const encoder = new TextEncoder();
  const data = encoder.encode(plainText);

  // RSA-OAEP로 암호화
  const encrypted = await crypto.subtle.encrypt(
    { name: 'RSA-OAEP' },
    publicKey,
    data
  );

  // Base64로 인코딩하여 반환
  return btoa(String.fromCharCode(...new Uint8Array(encrypted)));
};
⚠️ 주의: WebCrypto API (RSA-OAEP)를 사용하면 서버 측에서도 동일한 알고리즘(RSA/ECB/OAEPWithSHA-256AndMGF1Padding)을 사용해야 합니다. 기본 구현은 JSEncrypt 호환(RSA/ECB/PKCS1Padding)입니다.

9. 에러 코드 참조

코드 메시지 원인
3001 등록되지 않은 사용자입니다. 이메일로 사용자를 찾을 수 없음
3002 비밀번호가 일치하지 않습니다. 비밀번호 불일치
3003 계정이 잠겼습니다. 로그인 실패 횟수 초과
3004 비활성화된 계정입니다. 사용 중지된 계정
4001 RSA 키 로딩에 실패했습니다. 서버 RSA 키 설정 오류
4002 암호화된 데이터 복호화에 실패했습니다. 잘못된 암호화 데이터 또는 키 불일치
